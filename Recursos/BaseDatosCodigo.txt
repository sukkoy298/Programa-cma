CREATE TABLE Examenes(
ID_Alfa char(1) not null DEFAULT 'E',
ID_Num int(11) not null PRIMARY KEY auto_increment,
Name varchar(30) not null,
ValuePr float not null,
Costo float not null,
Disponible boolean not null
);
CREATE TABLE Reactivos_Necesarios(
ID_Reacti varchar(20) not null,
Cantidad_Requerida decimal,
ID_Examen varchar(20) not null
);
CREATE TABLE facturas(
ID_Alfa char(1) not null DEFAULT 'F',
ID_Num int(11) not null PRIMARY KEY auto_increment,
ID_Pacient varchar(20) not null,
Fecha date not null,
Siguiente_Pasar char(1) not null,
Costo float not null,
Tip_Servi tinytext not null
);
CREATE TABLE datos_extras_factura(
ID_Factura varchar(20) not null,
Dat_Es varchar(20) not null
);
CREATE TABLE historial_medico(
ID_Alfa char(1) not null DEFAULT 'H',
ID_Num int(11) not null PRIMARY KEY auto_increment,
ID_Factura varchar(20) not null
);
CREATE TABLE Resultados_Historial(
ID_Historial varchar(20) not null,
id int(11) not null Primary Key auto_increment,
Evaluacion varchar(30) not null,
resultado varchar(50) not null
);
CREATE TABLE datosVariosPacient(
ID_Pacient varchar(20) not null,
DatoVario varchar(50) not null,
clacificador varchar(20) not null
);
CREATE TABLE Reactivos(
ID_Alfa char(1) not null DEFAULT 'R',
ID_Num int(11) not null PRIMARY KEY auto_increment,
Name varchar(20) not null,
Existencia decimal not null,
Uni_Medida tinytext not null
);
CREATE TABLE Direcciones de los archivos(
nivel varchar(25) not null,
direcc varchar(40) not null
);
